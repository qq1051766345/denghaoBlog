    # 虚拟dom的的两种创建方式

    # 1.jsx形式创建

     <script type="text/babel">
    //1.创建虚拟dom
    const VDOM = <h1 id="title">hello react</h1>

    // 2.渲染虚拟dom到页面

  </script>

  </script>

        super(props);

          isHot:true,
          wind:'微风',
        };
        // 需要给类中的方法指定this
        //bind方法做了两件事，一件创建了changeWeather这个方法，二是改变了this指向
        this.changeWeather = this.changeWeather.bind(this);//这个方法挂在实例上
      }
      //这个方法挂在原型上
      changeWeather(){
        //changeWeather放在哪儿 --weather的原型对象上，供实例使用

        console.log(this.state.isHot);

        //this.state.isHot = !this.state.isHot;

        //  ！正确写法,状态必须通过setState进行更新，且是合并不是替换

        this.setState({
          isHot:!isHot,
        });
      }
      // render调用几次 1+n 次 1是初始化的那一次，n是状态更新的次数
      render(){
        const {isHot,wind} = this.state;
        return <h1 onClick={this.changeWeather}>今天天气很{isHot ? '炎热' : '凉爽'},{wind}</h1>
      }
      
    }

    # React生命周期

    # 1.生命周期（旧）

    生命周期图：

    1. 加载阶段
    2. 取得默认属性，初始状态在constructor中完成（运行一次，可读数据，可同步修改state，异步修改state需要setState,setState在实例产生后才可以使用，可以访问到props）
    3. 即将挂载 componentWillMount
    4. 描画VDOM render (创建虚拟dom，进行diff算法，更新dom树)
    5. 组件渲染之后调用 componentDidMount (一般在这个钩子中做一些初始化的事，例如：开启定时器、发送网络请求、订阅消息) 常用 更新阶段
    6. props改变 componentWillReceiveProps (组件加载时候不调用，组件接受新的props时调用)
    7. shouldComponentUpdate 是否需要更新（return true就会更新dom，false阻止更新）
    8. 描画VDOM render (创建虚拟dom，进行diff算法，更新dom树)
    9. componentDidUpdate 组件数据更新完成后调用
    10. 卸载阶段 compoenentWillUnmount (即将卸载，可以做一些组件相关的清理工作，如青春计时器、网络请求等 )常用
    11. 组件卸载： unmountComponentAtNode(document.getElementById('root'))
    12. 所有子组件挂载完成，才标记着父组件挂载完成，父组件更新，子组件更新，子组件更新，子组件不更新
    <script type="text/babel">
    //1.创建组件
    class Count extends React.Component{
      //构造器
      constructor(props){
        super(props);
        console.log('constructor');
      }

      //组件将要挂载
      componentWillMount(){
        console.log('componentWillMount');
      }

      //组件挂载完成的钩子
      componentDidMount(){
        console.log('componentDidMount');
      } 

      //组件即将被销毁的时候出发
      componentWillUnmount(){
        console.log('componentWillUnmount');
      }

      // 控制组件更新的阀门
      shouldComponentUpdate(){
        console.log('shouldComponentUpdata');
        return true;
      }

      // 组件将要更新
      componentWillUpdate(){
        console.log('componentWillUpdate');
      }

      // 组件更新完成
      componentDidUpdate(){
        console.log('componentDidUpdate');
      }

      // 父组件即将更新的时候触发
      componentWillReceiveProps(){
        console.log('count组件将要接收到新的props');
      }

      //初始化状态
      state= {
        count: 0
      }
      // 加一方法
      add = () => {
        //获取原来状态
        let {count} = this.state;
        //更新状态
        this.setState({count: count + 1});
      }

      // 卸载组件
      delete = () => {
        //卸载组件
        ReactDOM.unmountComponentAtNode(document.getElementById('app'));
      }

      // 强制更新组件
      force = () => {
        // 强制更新
        this.forceUpdate();
      }

      //创建组件
      render(){
        console.log('render');
        return (
          <div>
            <h2>当前A组件的车位{this.props.carName}</h2>
            <h2>当前求和为{this.state.count}</h2>

      }
    }

    # 2.生命周期（新）

    1. 在React16新的生命周期弃用了compostWillMount、componentWillReceiveProps、componentWillUpdate
    2. 新增了getDerivedStateFromProps、getSonapshotBeforeUpdate来替代弃用的三个钩子函数(compostWillMount、componentWillReceiveProps、componentWillUpdate)

    1. 渲染前 static getDerivedStateFromProps(props,state) 无法访问this props,state是更新后的 必须返回一个对象，用来更新state或者null 不更新 必须要初始化state 场景：state的值在任何时候都取决于props时
    2. render 必须return jsx|string|number|null 不会直接于浏览器交互：不要操作DOM｜和数据
    3. componentDidMount 挂载完成后执行 更新阶段
    4. 渲染前 static getDerivedStateFromProps(props,state) 无法访问this props,state是更新后的 必须返回一个对象，用来更新state或者null 不更新 必须要初始化state 场景：state的值在任何时候都取决于props时
    5. render 必须return jsx|string|number|null 不会直接于浏览器交互：不要操作DOM｜和数据
    6. getSnapshortBeforeUpdate(prevProps,prevState) 组件能在发生更改之前从DOM中捕获一些信息(dom渲染前的状态) 返回 值｜null 会给componentDidUpdate prevProps、prevState 更新前this.props、this.state更新后

    下列表是getSnapshotBeforeUpdate运用实例

    <script type="text/babel">
    class NewsList extends React.Component{
      //初始化状态
      state = {
        newsArr:[]
      }

      //更新前获取快照
      getSnapshotBeforeUpdate(){
        console.log(this.List.scrollHeight);
        //传给componentDidUpdate更新前盒子的高度
        return this.List.scrollHeight;
      }

      componentDidUpdate(preProps,preState,snapshot){
        //更新完成后，将滚动条的位置改变
        this.List.scrollTop += this.List.scrollHeight - snapshot;
        console.log(12313,this.List.scrollHeight - snapshot);
      }

      componentDidMount(){
        //挂载完成就开始一个循环定时器
        this.timer = setInterval(() => {
          //获取原状态
          let {newsArr} = this.state;
          //模拟一条新闻
          const news = '新闻' + (newsArr.length + 1);
          //将新闻添加到原状态中
          this.setState({
            // 合并数组
            newsArr:[news,...newsArr],
          })
        },1000)
      }
      render(){
        return (
          <div className="list" ref={k => this.List = k}>
            {
              this.state.newsArr.map(n => {
                return <div className="news" key={n}>{n}</div>
              })
            }
          </div>
        )
      }
    }

    //渲染组件
    ReactDOM.render(<NewsList/>,document.getElementById('app'));
  </script>

    # 2.state（钩子函数useState）

    state和props类似，都是一种存储属性的方式，但是不同点在于state只属于当前组件，其他组件无法访问。并且state是可变的，当其发生变化后组件会自动重新渲染，以使变化在页面中呈现。(类似于vue的data数据项但是需要自己手动设置)

    在函数中使用state我们需要使用一种钩子（hook）函数。钩子函数可以在函数组件中“勾出”React的特性，换句话说我们要用一个函数“勾出”state。

    const [state, setState] = useState(initialState);

    useState方法只有一个参数，就是当前state的默认值，方法会以数组的形式返回两个值：

    第一个值：是存储在state的变量

    第二个值：是操作此变量的函数

    # 3.ref（useRef）

    用react方法来获取dom对象

    从react中获取dom对象步骤：

    1.创建一个存储dom对象的容器

    使用useRef钩子函数

    钩子函数注意事项：

    1.钩子函数必须在函数组件的最外层

    2.钩子函数不能出现在条件语句中

    3.钩子函数不能出现在循环语句中

    4.钩子函数不能出现在嵌套函数中

    2.将容器设置为想要获取的dom元素的ref属性

    3.通过创建容器对象的current属性拿到对应的dom对象

      //1.创建容器  
  const h1Ref = useRef();

  const clickHandler = () => {
    console.log(h1Ref.current);
  }

  return (
    <div className='App'>
      {/* 2.设置容器 */}
      <h1 ref={h1Ref}>标题</h1>
      <button onClick={clickHandler}>1</button>
    </div>
  );

    useRef()所返回的对象就是一个普通的JS对象，所以上例中即使我们不使用钩子函数，仅仅创建一个形如{current:null}的对象也是可以的。只是我们自己创建的对象组件每次渲染时都会重新创建一个新的对象，而通过useRef()创建的对象可以确保组件每次的重渲染获取到的都是相同的对象。
